G_bloque1 = feedback(Gdirecto, H1, -1);  % Retroalimentaci贸n negativa
% Calcular G_bloque2 (Suma de G2 y G3)
G_bloque2 = G2 + G3;

G_cl = feedback(Gopen, H2, -1);  % Retroalimentaci贸n negativa
